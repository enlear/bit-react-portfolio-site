import { MemoryRouter } from 'react-router-dom';
import { ThemeProvider } from '@showoff/design.theme.theme-provider';
import { lightTheme } from '@showoff/design.theme.light-theme';
import { Contact } from './contact';

A component that renders a contact us page.

### Component usage

Use with `ThemeProvider` to apply themes.
Use with a React Router wrapper to enable routing.

```js
<MemoryRouter>
  <Contact />
</MemoryRouter>
```

### Render the contact us page on light theme.

```js live
<MemoryRouter>
  <ThemeProvider theme={lightTheme()}>
    <Contact />
  </ThemeProvider>
</MemoryRouter>
```
