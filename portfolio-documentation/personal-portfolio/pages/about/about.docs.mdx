import { MemoryRouter } from 'react-router-dom';
import { darkTheme } from '@showoff/design.theme.dark-theme';
import { Main } from '@showoff/personal-portfolio.layouts.main';
import { ThemeProvider } from '@showoff/design.theme.theme-provider';
import { About } from './about';

A component that renders an About Me Page

### Component usage

Use with a `ThemeProvider` for styling.
Use with a React Router to enable routing.

```js
<MemoryRouter>
  <About />
</MemoryRouter>
```

### Rendering the About Us page.

```js live
<MemoryRouter>
  <ThemeProvider theme={darkTheme()}>
    <Main>
      <About />
    </Main>
  </ThemeProvider>
</MemoryRouter>
```
