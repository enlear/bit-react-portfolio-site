import { MemoryRouter } from 'react-router-dom';
import { darkTheme } from '@showoff/design.theme.dark-theme';
import { ThemeProvider } from '@showoff/design.theme.theme-provider';
import { Appbar } from './appbar';

A component renders an app bar (navigation bar) for the portfolio

### Component usage

Use with `ThemeProvider` for a theme object.
Use with a React Router DOM implementation for the Router Links

```js
<Appbar />
```

### Render hello world!

```js live
<MemoryRouter>
  <ThemeProvider theme={darkTheme()}>
    <Appbar />
  </ThemeProvider>
</MemoryRouter>
```
