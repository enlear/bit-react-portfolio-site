import { ThemeProvider } from "@mui/material";
import { Picture } from "./picture";

A component that renders a User Avatar component for a site.
Use with `ThemeProvider` to enforce themeing.

### Component usage

Use with `ThemeProvider`

```js
<Picture pictureUrl={AVATAR_URL} />
```

### Render sample picture.

```js live
<ThemeProvider>
  <Picture
    pictureUrl={
      "https://images.unsplash.com/photo-1535713875002-d1d0cf377fde?ixlib=rb-4.0.3&ixid=MnwxMjA3fDB8MHxwaG90by1wYWdlfHx8fGVufDB8fHx8&auto=format&fit=crop&w=1180&q=80"
    }
  />
</ThemeProvider>
```
